Parses the source code of the files in @files to build uml classes. Returns an array containing all the parsed classes or nil if no ruby file were found in the @files array.

 
    def parse_sources!

      @source_files = @files.select {|f| f.match /\.rb/}
      return nil if @source_files.empty?

      @source_files.each do |file|
        puts "processing #{file}..."
        f = File.open file, 'r'
        (parse_file f).each {|c| @classes << c}
        f.close
      end

      @classes
    end
